# Text preprocessing function

def preprocess_text(text):
    lemmatizer = WordNetLemmatizer()
    tokens = word_tokenize(text)
    tokens = [lemmatizer.lemmatize(word.lower()) for word in tokens if word.lower() not in stopwords.words('english') and word.isalnum()]
    return ' '.join(tokens)
